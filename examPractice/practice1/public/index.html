function loadUsers_then() {
let url = '< url-address>';
fetch(url)
.then((response) => { return response.json() })
.then((data) => {
let str = `'< html-code>'`;
for (let curr of data.results) {
let str = `< html-code>${curr.key1.key2}`
}
document.querySelector('#mydiv').innerHTML = str;
})
}
